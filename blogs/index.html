package main

        import "fmt"
        import "math"

        func main() {
        	fmt.Printf("hello, world\n")
        	fmt.Printf("new line added\n")
        	fmt.Println(" result =", math.Float32bits(100))

        	fmt.Println("go" + "lang")

        	fmt.Println("1+1", 1+1)
        	fmt.Println("7.0/3.0 =", 7.0/3.0)

        	fmt.Println(true && false)
        	fmt.Println(true || false)
        	fmt.Printf("%t \n", !true)

        }
